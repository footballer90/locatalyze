Change 3 — Replace the "Format Output" Code Node

Find the Code node that builds the final competitor list. Replace its JavaScript with:

// ── Collect results from all 3 NearbySearch calls ────────────────────────────
const allPlaces = [];

for (let i = 0; i < $input.all().length; i++) {
  const item = $input.all()[i].json;
  const results = item.results || item.result || [];
  for (let j = 0; j < results.length; j++) {
    allPlaces.push(results[j]);
  }
}

// ── Deduplicate by place_id ───────────────────────────────────────────────────
const seen = {};
const unique = [];
for (let i = 0; i < allPlaces.length; i++) {
  const p = allPlaces[i];
  const key = p.place_id || p.name || ('idx-' + i);
  if (!seen[key]) {
    seen[key] = true;
    unique.push(p);
  }
}

// ── Haversine distance from search origin ─────────────────────────────────────
const originLat = parseFloat($('Parse Input').first().json.lat);
const originLng = parseFloat($('Parse Input').first().json.lng);

function haversineM(lat1, lng1, lat2, lng2) {
  const R = 6371000;
  const dLat = (lat2 - lat1) * Math.PI / 180;
  const dLng = (lng2 - lng1) * Math.PI / 180;
  const a = Math.sin(dLat/2) * Math.sin(dLat/2)
    + Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180)
    * Math.sin(dLng/2) * Math.sin(dLng/2);
  return Math.round(R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a)));
}

// ── Map to standard competitor shape ─────────────────────────────────────────
const competitors = unique.map(function(p) {
  const loc = p.geometry && p.geometry.location ? p.geometry.location : {};
  const distM = (loc.lat && loc.lng)
    ? haversineM(originLat, originLng, loc.lat, loc.lng)
    : null;

  return {
    name:            p.name || 'Unknown',
    address:         p.vicinity || p.formatted_address || null,
    rating:          p.rating || null,
    review_count:    p.user_ratings_total || null,
    // CRITICAL: pass google_types[] for compute engine type-based matching
    google_types:    p.types || [],
    place_id:        p.place_id || null,
    distance_m:      distM,
    // Tag as direct so engine trusts without re-checking keywords
    competitor_type: 'direct',
    business_status: p.business_status || null,
  };
}).filter(function(c) {
  return c.business_status !== 'PERMANENTLY_CLOSED';
});

// ── Final output ──────────────────────────────────────────────────────────────
const bizKey = $('Parse Input').first().json.bizKey;
const radius = $('Parse Input').first().json.radius;
const area   = $('Parse Input').first().json.area;
const city   = $('Parse Input').first().json.city;

const withinRadius = competitors.filter(function(c) {
  return c.distance_m !== null && c.distance_m <= radius;
});

return [{
  json: {
    // Primary key — what computeEngine reads
    competitors: competitors,

    // Metadata
    total_found:             allPlaces.length,
    unique_found:            unique.length,
    validated_count:         competitors.length,
    search_radius_m:         radius,
    biz_key:                 bizKey,
    area:                    area,
    city:                    city,
    data_source:             'google_places_nearbysearch',
    searched_at:             new Date().toISOString(),

    // Legacy fields for backward compat
    competitors_within_500m:   competitors.filter(function(c) { return c.distance_m !== null && c.distance_m <= 500; }).length,
    competitors_within_radius: withinRadius.length,
    saturation_level: competitors.length === 0 ? 'unknown'
                    : competitors.length <= 3  ? 'low'
                    : competitors.length <= 8  ? 'moderate'
                    : 'high',
  }
}];
⚠️ Replace $('Parse Input') with the actual name of your Parse node if it's named differently (e.g. $('Set Variables') or $('Code')).

What the Compute Engine Now Does With This Data

  1. Reads a1.competitors first, plus 9 other field name variants as fallback
  2. Calls isValidCompetitor(place, bizKey) which checks google_types[] against known API types before keyword scan
  3. Since all results are tagged competitor_type: 'direct', they pass validation immediately
  4. Sets competitorDataQuality = 'live_verified' when ≥1 competitor found
  5. Only awards competition score 85+ (blue ocean) when data quality is live_verified AND count is 0 — so a data failure can never masquerade as an opportunity
